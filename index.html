    <script>

        const tg = window.Telegram.WebApp;

        tg.expand(); // Раскрываем приложение на весь экран


        // ВНИМАНИЕ: Замени этот URL на свой адрес Webhook из n8n (Шаг 1)

        const WEBHOOK_URL = https://kreha.ru/webhook-test/expedition-location;


        const btn = document.getElementById('sendBtn');

        const status = document.getElementById('status');

        const title = document.getElementById('title');


        btn.onclick = () => {

            btn.disabled = true;

            status.innerText = "Определяю точные координаты GPS...";

            

            if (!navigator.geolocation) {

                status.innerText = "Ошибка: Ваше устройство не поддерживает передачу GPS.";

                btn.disabled = false;

                return;

            }


            navigator.geolocation.getCurrentPosition((pos) => {

                const data = {

                    user_id: tg.initDataUnsafe.user?.id || 0,

                    lat: pos.coords.latitude,

                    lon: pos.coords.longitude

                };


                status.innerText = "Передача данных в штаб n8n...";


                fetch(WEBHOOK_URL, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify(data)

                })

                .then(response => {

                    if (response.ok) {

                        status.style.color = "#4caf50";

                        status.innerText = "Координаты успешно сохранены! Хорошей прогулки.";

                        title.innerText = "Готово!";

                        tg.HapticFeedback.notificationOccurred('success');

                        setTimeout(() => tg.close(), 1500);

                    } else {

                        throw new Error('Ошибка сервера');

                    }

                })

                .catch((err) => {

                    status.style.color = "#ff5252";

                    status.innerText = "Ошибка связи: " + err.message;

                    btn.disabled = false;

                });

            }, (err) => {

                status.style.color = "#ff5252";

                status.innerText = "Ошибка доступа к GPS. Проверьте разрешения в настройках браузера или Telegram.";

                btn.disabled = false;

            }, {

                enableHighAccuracy: true,

                timeout: 10000,

                maximumAge: 0

            });

        };
